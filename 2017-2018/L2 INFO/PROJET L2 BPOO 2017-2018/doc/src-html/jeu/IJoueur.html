001package jeu;
002
003import java.util.ArrayList;
004
005/**
006 * Cette interface définit le comportement (normalement) minimal d'un joueur
007 *
008 */
009public interface IJoueur {
010        /**
011         * Nombre max de mana que le joueur peut avoir (10 en principe)
012         */
013        public final static int MAX_MANA = 10;          // Nombre max de mana
014        
015        /**
016         * Le nombre max de cartes qui peuvent composer un deck. 
017         */
018        public final static int TAILLE_DECK = 15;       // Nombre max de cartes dans le deck
019
020        /**
021         * @return C'est le héros choisi par le joueur
022         */
023        Heros getHeros();
024        
025        /**
026         * @return C'est le pseudo du joueur
027         */
028        String getPseudo();
029
030        /**
031         * Le nombre de mana augmente de 1 à chaque tour. Mais il ne pourra pas dépasser la constante MAX_MANA.
032         * Certaines cartes peuvent augmenter ce nombre en cours de jeu (comme la carte Croissance sauvage)
033         * @return nombre de manas potentiels du joueur au début du tour
034         */
035        int getMana();
036        
037        /**
038         * Nombre de manas encore disponibles pendant le tour. Le stock est rempli à chaque début de tour avec les mans potentiels.
039         * @return nombre de mana encore en stock
040         * @see getMana() 
041         */
042        int getStockMana();
043        
044        /**
045         * La main est l'ensemble des cartes piochées par le joueur mais non encore mises en jeu.
046         * @return La liste des cartes de la main
047         */
048        ArrayList<ICarte> getMain();
049        
050        /**
051         * Le jeu d'un joueur est composé des cartes placées sur le board
052         * @return La liste des cartes en jeu
053         */
054        ArrayList<ICarte> getJeu();
055        
056        /**
057         * Permet de rechercher une carte en jeu par rapport à un bout de son nom (grâce à la fonction contains de la classe String).
058         * @param nomCarte C'est une chaine de caractères qui correspond au nom ou à un bout du nom de la carte que l'on souhaite
059         * @return la première carte trouvée ou null si aucune carte ne correspond
060         */
061        ICarte getCarteEnJeu(String nomCarte);  
062        
063        
064        /**
065         * Permet de rechercher une carte de la main par rapport à un bout de son nom (grâce à la fonction contains de la classe String).
066         * @param nomCarte C'est une chaine de caractères qui correspond au nom ou à un bout du nom de la carte que l'on souhaite
067         * @return la première carte trouvée ou null si aucune carte ne correspond
068         */
069        ICarte getCarteEnMain(String nomCarteMain);
070
071        
072        
073        /**
074         * Il se passe beaucoup de choses au moment où le joueur prend le tour. Son nombre de mana augmente,
075         * son stock est réinitialisé en conséquence, les erviteurs en attente deviennent jouables, etc.
076         * @throws HearthstoneException Si jamais le joueur tente de prendre le tour alors que ce n'est pas à lui de le faire...
077         */
078        void prendreTour() throws HearthstoneException;
079
080        /**
081         * Il se passe beaucoup de choses au moment où le joueur finit le tour. 
082         * @throws HearthstoneException Si jamais le joueur tente de finir le tour alors que ce n'est pas à lui de le faire...
083         */
084        void finirTour() throws HearthstoneException;
085        
086        /**
087         * Au début de chaque tour, une carte du deck est choisie aléatoirement et est placée dans la main du joueur. 
088         * Elle est naturellement enlevée de son deck.
089         * @throws HearthstoneException En cas de problème, une exception...
090         */
091        void piocher() throws HearthstoneException;
092        
093        /**
094         * Pendant le tour, un joueur peut tenter de prendre une carte de sa main et de la placer sur le board. 
095         * Si c'est possible, elle est retirée de la main et placée sur le board. Evidemment, il faut faire
096         * attention aux effets de mise en jeu éventuels, aux contrôles, etc.
097         * @param carte La carte jouée
098         * @throws HearthstoneException En cas de problème, une exception...
099         * @see ICarte
100         */
101        void jouerCarte(ICarte carte) throws HearthstoneException;
102        
103        /**
104         * Pendant le tour, un joueur peut tenter de prendre une carte de sa main et de la placer sur le board. 
105         * Si c'est possible, elle est retirée de la main et placée sur le board. Evidemment, il faut faire
106         * attention aux effets de mise en jeu éventuels, aux contrôles, etc. C'est la même chose que la fonction
107         * jouerCarte(carte), mais ici il faut préciser la cible obligatoirement
108         * @param carte La carte...
109         * @param cible Sa cible...
110         * @throws HearthstoneException En cas de problème, une exception...
111         */
112        void jouerCarte(ICarte carte, Object cible) throws HearthstoneException;
113        
114        /**
115         * Lorsque qu'une carte est en jeu, le joueur peut utiliser cette carte. Utiliser un serviteur
116         * consistera à attaque la cible (un autre serviteur ou le héros), à utiliser un sort, etc. 
117         * @param carte La carte
118         * @param cible La cible
119         * @throws HearthstoneException En cas de problème, une exception...
120         */
121        void utiliserCarte(ICarte carte, Object cible) throws HearthstoneException;
122        
123        /**
124         * Cette fonction utilise le pouvoir du héros du joueur pour faire quelque chose sur la cible...
125         * @param cible La cible...
126         * @throws HearthstoneException En cas de problème, une exception...
127         */
128        void utiliserPouvoir(Object cible) throws HearthstoneException;
129        
130        /**
131         * Lorsqu'une carte a fini sa vie (un serviteur qui meut, un sort qui vient d'être utilisé, ...), il 
132         * faut gérer le jeu (une carte en moins...), ainsi que les effets liés à la disparition de la carte (Rale d'agonie par exemple).
133         * @param carte La carte qui est perdue
134         * @throws HearthstoneException En cas de problème, une exception...
135         */
136        void perdreCarte(ICarte carte) throws HearthstoneException;
137        
138        
139        
140}
