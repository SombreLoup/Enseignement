001package jeu;
002
003/**
004 * Une carte, ben c'est une carte du jeu quoi ! Une carte peut être
005 * une carte de sort ou une carte serviteur (on ne s'occupe pas des autres sortes de cartes).
006 * @see IJoueur
007 */
008public interface ICarte extends Cloneable {
009        
010        String getNom();                                // Une carte doit avoir un nom
011        int getCout();                          // Une carte coûte un crtain nombre de mana
012        IJoueur getProprietaire();      // Une carte appartient à un joueur
013        
014        /**
015         * Une carte peut avoir un effet au début de chaque tour où elle est en jeu
016         * @param cible ce parametre peut être null si la carte n'a pas besoin d'une cible pour l'effet en question. 
017         * La cible peut être égale à n'importe quoi d'autre qui arrange la carte (un héros, un serviteur, une autre carte...) 
018         * @throws HearthstoneException
019         */
020        void executerEffetDebutTour(Object cible) throws HearthstoneException;
021
022        /**
023         * Une carte peut avoir un effet à la fin d'un chaque tour où elle est en jeu
024         * @param cible ce parametre peut être null si la carte n'a pas besoin d'une cible pour l'effet en question.
025         * La cible peut être égale à n'importe quoi d'autre qui arrange la carte (un héros, un serviteur, une autre carte...) 
026         * @throws HearthstoneException
027         */
028        void executerEffetFinTour(Object cible) throws HearthstoneException;
029        
030        /**
031         * Une carte peut avoir un effet au début de sa mise en jeu 
032         * @param cible ce parametre peut être null si la carte n'a pas besoin d'une cible pour l'effet en question. 
033         * La cible peut être égale à n'importe quoi d'autre qui arrange la carte (un héros, un serviteur, une autre carte...)   * @throws HearthstoneException
034         */
035        void executerEffetDebutMiseEnJeu(Object cible) throws HearthstoneException;
036        
037        /**
038         * Une carte peut avoir un effet au moment de sa disparition du jeu 
039         * @param cible ce parametre peut être null si la carte n'a pas besoin d'une cible pour l'effet en question. 
040         * La cible peut être égale à n'importe quoi d'autre qui arrange la carte (un héros, un serviteur, une autre carte...)   * @throws HearthstoneException
041         * @throws HearthstoneException
042         */
043        void executerEffetDisparition(Object cible) throws HearthstoneException;
044        
045        /**
046         * Une carte peut avoir une action qui se commande à n'importe quel moment du tour lorsqu'elle est en jeu 
047         * @param cible ce parametre peut être null si la carte n'a pas besoin d'une cible pour l'action en question. 
048         * La cible peut être égale à n'importe quoi d'autre qui arrange la carte (un héros, un serviteur, une autre carte...)   * @throws HearthstoneException
049         * @throws HearthstoneException
050         */
051        void executerAction(Object cible) throws HearthstoneException;
052        
053        /**
054         * Fonction qui teste si les conditions pour que la carte soit encore présente au tour suivant. Si la fonction
055         * renvoie vrai, il faut sûrement la retirer du board...
056         * @return true si la carte est foutu (un serviteur tué, un sort lancé, etc.)
057         */
058        boolean disparait();
059}
